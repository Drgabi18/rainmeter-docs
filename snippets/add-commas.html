Returns an input number as an output string formatted with commas.

{% highlight lua %} function FormatCommas(numVar) assert(tonumber(numVar), 'Argument '..numVar..' is invalid. FormatCommas() expects a number.') local prefix, number, postfix = string.match(numVar, '^([^%d]*%d)(%d*)(.-)$') return prefix..(number:reverse():gsub('(%d%d%d)', '%1,'):reverse())..postfix end {% endhighlight %}
