These functions can be used to change a string to either Sentence Case or Camel Case.

{% highlight lua %} function SentenceCase(input) assert(type(input) == 'string', ('SentenceCase must receive a string. Received %s instead.'):format(type(input))) return (input:gsub('[^.]+', function(sentence) local space, first, rest = sentence:match('(%s*)(.)(.*)') return space .. first:upper() .. rest:lower():gsub('%si%s', ' I ') end)) end -- SentenceCase {% endhighlight %}
{% highlight lua %} function CamelCase(input) assert(type(input) == 'string', ('CamelCase must receive a string. Received %s instead.'):format(type(input))) return (input:gsub('%w+', function(word) local first, rest = word:match('(.)(.*)') return first:upper() .. rest:lower() end)) end -- CamelCase {% endhighlight %}
